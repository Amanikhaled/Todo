  DeleteItem(id: number) {
    const index = this.list.findIndex(rl => rl.id === id)
    this.deletedTodo.push(this.list[index])
    this.list.splice(this.list.findIndex(el => el.id == id), 1)

  Done(item: Todo) {
    item.completed = !item.completed;
  }
  getAllTodos(): Todo[] {
    return this.list;

  }
  AddToFAvorite(item: string) {
    const index = this.list.findIndex(e => e.todo === item)
    if (index != -1) {
      this.list[index].favorite = !this.list[index].favorite;
    }
    this.favorite.push(this.list[index]);
  }
  getAllFavorite() {
    return this.favorite;
  }

    return count;
  }

  CompletedTodo() {
    const completedTodos = this.list.filter(todo => todo.completed === false);

    const percentage = Math.round((completedTodos.length / this.list.length) * 100).toFixed(2);
    return percentage || 0;
